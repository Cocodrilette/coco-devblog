  REFERENCIA PARA LAS CLASES EN CSS
//---------------------------------//

Contenedor principal (contiene el header y la navbar) => 
	[div] mainContainer   

//---------------------------------//

Contenedor dentro del principal(contiene solo el contenido) =>
	[div] mainContent

//---------------------------------//

Titulo de la página =>
	[h1] mainTitle

//---------------------------------//

Header de la página => 
	[header] = header

//---------------------------------//

Contenedor dentro del header =>
	[div] headerContent

//---------------------------------//

Barra de navegación =>
	[nav] navbarContainer

//---------------------------------//

Nombre de la marca => 
	[div] navbarBrand

//---------------------------------//

navbar Item =>
	[li] navbarItem

//---------------------------------//

Link del navbar item =>
	[Link] navbarItemLink

//---------------------------------//

Imagen principal del home => 
	[figure] imageSection

//---------------------------------//

Creditos para la imagen principal =>
	[figcaption] => ImageCaption

//---------------------------------//

card de post toda la pantalla=> 
	[div] postCardLg

//---------------------------------//

titulo de la carta LG =>
	[h3] cardTitleLg

//---------------------------------//

informacion de la card LG => 
	[div] cardInfoLg

//---------------------------------//

parrafo de la card LG =>
	[p] cardParaLg

//---------------------------------//

descripcion de la card LG =>
	[p] cardDescriptionLg

//---------------------------------//

section con información sobre un post =>
	[section] infoSection

//---------------------------------//

parrafo de section de información =>
	[p] infoSectionPara

//---------------------------------//

nombre del author en seccion de informacion => [span > a] infoSectionParaAuthor

//---------------------------------//

contenido del post =>
	[div] postContent

//---------------------------------//

footer del post => 
	[footer] postFooter
